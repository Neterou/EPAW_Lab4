-- ---------- Bubbles: hyper-local community channels ----------
CREATE TABLE IF NOT EXISTS bubbles (
  id        INTEGER PRIMARY KEY AUTOINCREMENT,
  name      VARCHAR(100) NOT NULL,
  category  VARCHAR(20)  NOT NULL DEFAULT 'community',  -- sports|community|public|transport|special
  zip       VARCHAR(5)   NOT NULL,
  city      VARCHAR(100) NOT NULL,
  country   VARCHAR(2)   NOT NULL,
  lat       REAL         NOT NULL DEFAULT 0,
  lng       REAL         NOT NULL DEFAULT 0,
  open      INTEGER      NOT NULL DEFAULT 1,            -- 1 = anyone joins, 0 = approval required
  owner_id  INTEGER,                                    -- creator/owner who approves join requests (backfilled by DBManager)
  created_at TIMESTAMP   DEFAULT CURRENT_TIMESTAMP
);

-- ---------- Users: full registration profile ----------
CREATE TABLE IF NOT EXISTS users (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  name          VARCHAR(60)  NOT NULL,
  email         VARCHAR(255) NOT NULL UNIQUE,
  phone         VARCHAR(15),
  phone_country VARCHAR(5),
  dni           VARCHAR(9)   NOT NULL UNIQUE,
  zip           VARCHAR(5)   NOT NULL,
  city          VARCHAR(100) NOT NULL,
  country       VARCHAR(2)   NOT NULL,
  gender        VARCHAR(20),
  password      VARCHAR(255) NOT NULL,
  username      VARCHAR(30)  NOT NULL UNIQUE,
  picture       VARCHAR(255),
  status        VARCHAR(10)  NOT NULL DEFAULT 'PENDING', -- PENDING|APPROVED|BANNED
  role          VARCHAR(20)  NOT NULL DEFAULT 'USER',    -- USER|ADMIN
  bubble_id     INTEGER,
  created_at    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (bubble_id) REFERENCES bubbles(id)
);

-- ---------- Memberships: a user can belong to many bubbles ----------
CREATE TABLE IF NOT EXISTS user_bubbles (
  user_id   INTEGER NOT NULL,
  bubble_id INTEGER NOT NULL,
  status    VARCHAR(10) NOT NULL DEFAULT 'APPROVED',  -- APPROVED | PENDING
  joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (user_id, bubble_id),
  FOREIGN KEY (user_id)   REFERENCES users(id)   ON DELETE CASCADE,
  FOREIGN KEY (bubble_id) REFERENCES bubbles(id) ON DELETE CASCADE
);

-- ---------- Tweets: posts broadcast inside a bubble ----------
CREATE TABLE IF NOT EXISTS tweets (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id    INTEGER NOT NULL,
  bubble_id  INTEGER NOT NULL,
  content    VARCHAR(280) NOT NULL,
  type       VARCHAR(12) NOT NULL DEFAULT 'STANDARD',   -- STANDARD|OFFICIAL|ALERT|WARNING
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id)   REFERENCES users(id)   ON DELETE CASCADE,
  FOREIGN KEY (bubble_id) REFERENCES bubbles(id) ON DELETE CASCADE
);

-- ---------- Follows: user <-> user relationship ----------
CREATE TABLE IF NOT EXISTS follows (
  follower_id INTEGER NOT NULL,
  followee_id INTEGER NOT NULL,
  followed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (follower_id, followee_id),
  FOREIGN KEY (follower_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (followee_id) REFERENCES users(id) ON DELETE CASCADE,
  CHECK (follower_id <> followee_id)
);

-- ---------- Comments: threaded replies on a tweet ----------
CREATE TABLE IF NOT EXISTS comments (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  tweet_id   INTEGER NOT NULL,
  user_id    INTEGER NOT NULL,
  content    VARCHAR(500) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (tweet_id) REFERENCES tweets(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id)  REFERENCES users(id)  ON DELETE CASCADE
);

-- ---------- Likes: one reaction per user per tweet ----------
CREATE TABLE IF NOT EXISTS likes (
  user_id    INTEGER NOT NULL,
  tweet_id   INTEGER NOT NULL,
  reacted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (user_id, tweet_id),
  FOREIGN KEY (user_id)  REFERENCES users(id)  ON DELETE CASCADE,
  FOREIGN KEY (tweet_id) REFERENCES tweets(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_tweets_bubble   ON tweets(bubble_id);
CREATE INDEX IF NOT EXISTS idx_comments_tweet  ON comments(tweet_id);
CREATE INDEX IF NOT EXISTS idx_likes_tweet     ON likes(tweet_id);
CREATE INDEX IF NOT EXISTS idx_follows_followee ON follows(followee_id);
CREATE INDEX IF NOT EXISTS idx_memberships_bubble ON user_bubbles(bubble_id)
